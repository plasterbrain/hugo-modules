{{- /* MODULE: SEO Function: get-ogCover @requires hugo-modules/colors --- Creates a basic Open Graph cover for the given page. @NOTE DOES NOT WORK WITH AVIF --- Params: - .Page - .Cover --- Returns - [string] Path to the page's generated OG cover image */ -}} {{- $page := .Page -}} {{- $cover := .Cover -}} {{- $useCover := and $cover (eq (printf "%T" $cover) "*resources.resourceAdapter") -}} {{- $size := string site.Params.seo.OGCoverTemplate.Size | default "1200x630" -}} {{- $bg := partialCached "functions/get-themeBg.html" . "default" -}} {{- /* Overlay template on our cover image, if any */ -}} {{- $overlay := partialCached "functions/get-ogCoverTemplateOverlay.html" . "default" -}} {{- if $useCover -}} {{- with $overlay -}} {{- $cover = $cover.Filter (images.Overlay $overlay 0 0) -}} {{- end -}} {{- else -}} {{- with resources.Get site.Params.Seo.OGCover -}} {{- $cover = . -}} {{- else -}} {{- /* Theme color bg with text. */ -}} {{- with $bg -}} {{- $textColor := "#ffffff" -}} {{- with site.Params.Seo.ThemeColor -}} {{- $textColor = partialCached "functions/get-wcagBW.html" (dict "Color" .) . -}} {{- end -}} {{- $title := $page.LinkTitle | plainify -}} {{- $cover = $bg.Filter (images.Text (string $title) (dict "color" $textColor "size" 60 "linespacing" 2 "x" (div $bg.Width 3) "y" (div $bg.Height 2.5) )) -}} {{- end -}} {{- end -}} {{- end -}} {{- $cover = $cover.Crop (print $size " topleft") -}} {{- /* Rename the image */ -}} {{- $path := site.Params.Seo.OGCoverTemplate.ImageDir | default "images" -}} {{- $permalink := (path.Join $path (printf "%s.jpg" ($page.LinkTitle | anchorize))) -}} {{- $cover = resources.Copy $permalink $cover -}} {{- return $cover -}}

{{- /* MODULE: EMBEDS (Function) get-icon --- Fetches, executes, and returns a given icon in `assets/icons`. --- Params: - .Icon Icon basename, ex. "twitter" for "twitter.svg" */ -}} {{- $svg := false -}} {{- with resources.Get (printf "icons/%s.svg" .Icon) -}} {{- $svg = . | resources.ExecuteAsTemplate "icon.svg" . -}} {{- $svg = $svg.Content -}} {{- end -}} {{- return $svg -}}
